from starlette.middleware.base import BaseHTTPMiddleware
from starlette.responses import Response as StarletteResponse

class LocaleCookieMiddleware(BaseHTTPMiddleware):
    """Sets preferred_locale cookie on every /{locale}/... response."""

    async def dispatch(self, request, call_next):
        from app.lib.i18n import SUPPORTED_LOCALES
        response = await call_next(request)
        path_parts = request.url.path.lstrip("/").split("/")
        if path_parts and path_parts[0] in SUPPORTED_LOCALES:
            response.set_cookie(
                "preferred_locale",
                path_parts[0],
                max_age=365 * 24 * 3600,
                httponly=False,
                samesite="lax",
            )
        return response

app.add_middleware(LocaleCookieMiddleware)
```

Place `app.add_middleware(LocaleCookieMiddleware)` **before** `app.add_middleware(SessionMiddleware, ...)` (middlewares run in reverse registration order in Starlette — we want cookie set after session middleware processes, so register it after SessionMiddleware).

## Task 4: Fix player recent-game stats showing 0 (Bug 2)

**Files:**
- Modify: `backend/app/services/local_stats_aggregator.py`
- Test: `backend/tests/test_local_stats_aggregator.py`

**Root cause:** `backfill_game_player_stats_from_events` marks a game ineligible if ANY player already has `goals > 0`. This means games with partially resolved events (some players got goals, others failed name resolution) are never re-processed. The players with failed name resolution stay at 0 forever.

**Fix:** Also include games that have unresolved `UnresolvedPlayerEvent` rows (`resolved_at IS NULL`) — these games need a second pass with potentially improved name resolution.

def search_league_scorers(db_league_id: int, query: str, limit: int = 50) -> list[dict]:
    """Search scorers in a league by player name substring.

    If query is empty, returns top scorers ordered by points (same as get_league_top_scorers).
    Returns list of dicts with keys: player_id, player_name, team_name, gp, g, a, pts, pim.
    """
    import re as _re

    db = get_database_service()
    with db.session_scope() as session:
        league = session.query(League).filter(League.id == db_league_id).first()
        if league is None:
            return []

        league_abbrev = _re.sub(
            r"^(Junioren/-innen|Junioren|Juniorinnen|Herren|Damen|Senioren)\s+",
            "",
            str(league.name or ""),
        ).strip()

        base_q = (
            session.query(PlayerStatistics, Player)
            .join(Player, PlayerStatistics.player_id == Player.person_id)
            .filter(
                PlayerStatistics.season_id == league.season_id,
                PlayerStatistics.league_abbrev == league_abbrev,
            )
        )
        if league.game_class:
            base_q = base_q.filter(
                (PlayerStatistics.game_class == league.game_class)
                | (PlayerStatistics.game_class == None)  # noqa: E711
            )

        q = query.strip().lower()
        if q:
            base_q = base_q.filter(
                func.lower(Player.first_name + " " + Player.last_name).contains(q)
            )

        rows = (
            base_q
            .order_by(PlayerStatistics.points.desc(), PlayerStatistics.goals.desc())
            .limit(limit)
            .all()
        )

        return [
            {
                "player_id": ps.player_id,
                "player_name": f"{pl.first_name or ''} {pl.last_name or ''}".strip(),
                "team_name": ps.team_name or "",
                "gp": ps.games_played or 0,
                "g": ps.goals or 0,
                "a": ps.assists or 0,
                "pts": ps.points or 0,
                "pim": ps.penalty_minutes or 0,
            }
            for ps, pl in rows
        ]

            session.query(PlayerStatistics, Player)
            .join(Player, PlayerStatistics.player_id == Player.person_id)
            .filter(
                PlayerStatistics.season_id == league.season_id,
                PlayerStatistics.league_abbrev == league_abbrev,
                (PlayerStatistics.penalty_minutes or 0) > 0,
            )
        )
        if league.game_class:
            base_q = base_q.filter(
                (PlayerStatistics.game_class == league.game_class)
                | (PlayerStatistics.game_class == None)  # noqa: E711
            )

        q = query.strip().lower()
        if q:
            base_q = base_q.filter(
                func.lower(Player.first_name + " " + Player.last_name).contains(q)
            )

        rows = (
            base_q
            .order_by(PlayerStatistics.penalty_minutes.desc())
            .limit(limit)
            .all()
        )

        return [
            {
                "player_id": ps.player_id,
                "player_name": f"{pl.first_name or ''} {pl.last_name or ''}".strip(),
                "team_name": ps.team_name or "",
                "gp": ps.games_played or 0,
                "pim": ps.penalty_minutes or 0,
                "pen_2": ps.pen_2 if hasattr(ps, 'pen_2') else 0,
                "pen_5": ps.pen_5 if hasattr(ps, 'pen_5') else 0,
                "pen_10": ps.pen_10 if hasattr(ps, 'pen_10') else 0,
                "pen_match": ps.pen_match if hasattr(ps, 'pen_match') else 0,
            }
            for ps, pl in rows
        ]
```

Make sure `func` is imported at the top of `stats_service.py` (it's already used elsewhere — confirm with `grep "from sqlalchemy import\|sqlalchemy.sql.func" stats_service.py`).
